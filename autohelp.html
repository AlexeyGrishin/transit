  install: (transit) ->
    handler = (req, res) ->
      help = [options.greeting + ": "].concat(req.handlers.map((h) -> h.pattern).filter(_.identity)).join("\n - ")
      res.sendBack help
    transit.receive options.command, handler
    transit.receive handler if options.showOnUnknown
